CREATE TABLE STUD_TBL
    (STUD_ID VARCHAR(50) PRIMARY KEY,
    STUD_NAME VARCHAR2(100) NOT NULL,
    STUD_ADDRESS  VARCHAR2(1000),
    STUD_CONTACT_NO NUMBER,
    STUD_STANDARD_ID NUMBER,
    FOREIGN KEY(STUD_STANDARD_ID) REFERENCES COURSES_TBL(COURSE_ID) ON DELETE SET NULL,
    STUD_DIV CHAR,
    PARENT_NAME VARCHAR2(100),
   P_CONTACT_NO NUMBER,
  P_OFFICE_ADD VARCHAR2(1000),
   P_MOBILE_NO NUMBER,
   P_EMAIL_ID VARCHAR2(300));

CREATE TABLE COURSES_TBL
	(COURSE_ID NUMBER PRIMARY KEY,
	COURSE_NAME VARCHAR2(100),
        COURSE_TYPE VARCHAR2(100),
	COURSE_DIV CHAR,
	COURSE_FEES NUMBER);


CREATE TABLE STUD_FEES_DETAILS(
    STUD_ID VARCHAR(50),
    FOREIGN KEY(STUD_ID) REFERENCES STUD_TBL(STUD_ID) ON DELETE SET NULL,
    FEES_PAID NUMBER,
   FEES_BALANCE NUMBER,
    TOTAL_FEES NUMBER,
    PAYMENT_DATE NUMBER,
    DUE_DATE DATE);


CREATE TABLE FEES_TABLE_AUDIT(

    USER_NAME VARCHAR(40),
    TYPE_OF_OPERATION CHAR(30),

   DATE_TIME VARCHAR2(30));
